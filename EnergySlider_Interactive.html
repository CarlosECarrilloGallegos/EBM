Real-Time Energy-Balance Model (Interactive)¶

%matplotlib widget
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.widgets import Slider, Button
import matplotlib.animation as animation
import random

#constants
C = 2.08e8
alpha = 0.32
epsilon = 1 #key value
sigma = 5.67e-8
S = 1350


#step array
step = []
flux = []
#index = count()

fig = plt.figure()
ax1 = fig.add_subplot(1,1,1)
plt.title('Energy Balance Model')
plt.subplots_adjust(left=0.25, bottom=0.25)

#Sliders
axflux = plt.axes([0.1, 0.25, 0.0225, 0.63])
slider_flux = Slider(
    ax=axflux,
    label="Flux",
    valmin=1200,
    valmax=1400,
    valinit=1350,
    orientation="vertical"
)

axalpha = plt.axes([0.25, 0.05, 0.65, 0.03])
slider_alpha = Slider(
    ax=axalpha,
    label="Albedo",
    valmin=0.2,
    valmax=0.4,
    valinit=0.3,
    orientation="horizontal"
)

axemis = plt.axes([0.25, 0.1, 0.65, 0.03])
slider_emis = Slider(
    ax = axemis,
    label = "Emissivity",
    valmin = 0.6,
    valmax = 0.8,
    valinit = 0.7,
    orientation= "horizontal"
)

def animate(i):
    step.append(i)
    flux.append((((1 - slider_alpha.val)*(slider_flux.val)/(4*slider_emis.val*sigma))**(1/4)))
    ax1.set_ylim(250,320)
    ax1.plot(step, flux, color = 'green')
    ax1.set_xlabel('Step')
    ax1.set_ylabel('Temperature (K)')


ani = animation.FuncAnimation(fig, animate, interval = 80)
plt.show()
